▶ 30 秒上手›
  1. 在左侧「武将属性」填 id、name、hp、atk、def、移动/攻击范围。
  2. 在右侧「技能列表」点「+ 添加」，写 filter 与 content 代码。
  3. 底部输入密码 diy123，点「提交到服务器」。
  4. 回到游戏首页，DIY 武将会出现在选将池中。
▶ 主动技能示例›

filter：return actor && actor.alive && !actor.skilled;

const target = await Effect.chooseEnemy(actor, {
  range: { shape: "+", n: 3 },
  hintText: "请选择敌人"
});
if (!target) return false;
actor.skilled = true;
Effect.damage(actor, target, Effect.getEffectiveAttack(actor), { mul: 1.5 });
global.Game.log(actor.name + " 发动技能！");
return true;
